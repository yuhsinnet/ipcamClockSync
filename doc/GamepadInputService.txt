using System;
using System.Threading.Tasks;
using System.Runtime.InteropServices;
using System.Windows.Threading;

namespace IPCAM_Manger.Services
{
    public enum GamepadButton
    {
        A,
        B,
        LeftShoulder,
        RightShoulder,
        LeftTrigger,
        RightTrigger
    }

    public sealed class GamepadInputService : IDisposable
    {
        private const int ErrorSuccess = 0;
        private const ushort ButtonA = 0x1000;
        private const ushort ButtonB = 0x2000;
        private const ushort ButtonLeftShoulder = 0x0100;
        private const ushort ButtonRightShoulder = 0x0200;
        private const byte TriggerThreshold = 30;

        private readonly DispatcherTimer _pollTimer;
        private bool _isDisposed;
        private bool _isConnected;
        private ushort _lastButtons;
        private byte _lastLeftTrigger;
        private byte _lastRightTrigger;

        public const ushort WeakStrength = 10000;

        public GamepadInputService(int pollIntervalMilliseconds = 60)
        {
            _pollTimer = new DispatcherTimer
            {
                Interval = TimeSpan.FromMilliseconds(Math.Max(20, pollIntervalMilliseconds))
            };
            _pollTimer.Tick += PollTimer_Tick;
        }

        public event EventHandler<bool>? ConnectionChanged;

        public event EventHandler<GamepadButton>? ButtonPressed;

        public bool IsConnected => _isConnected;

        public void Start()
        {
            ThrowIfDisposed();
            _pollTimer.Start();
            PollOnce();
        }

        public void Stop()
        {
            if (_isDisposed)
            {
                return;
            }

            _pollTimer.Stop();
        }

        public void Dispose()
        {
            if (_isDisposed)
            {
                return;
            }

            _pollTimer.Stop();
            _pollTimer.Tick -= PollTimer_Tick;
            SetVibration(0, 0);
            _isDisposed = true;
        }

        private void PollTimer_Tick(object? sender, EventArgs e)
        {
            PollOnce();
        }

        private void PollOnce()
        {
            int result = XInputGetState(0, out XInputState state);
            bool connected = result == ErrorSuccess;

            if (connected != _isConnected)
            {
                _isConnected = connected;
                _lastButtons = 0;
                _lastLeftTrigger = 0;
                _lastRightTrigger = 0;
                ConnectionChanged?.Invoke(this, _isConnected);
            }

            if (!connected)
            {
                return;
            }

            ushort currentButtons = state.Gamepad.Buttons;
            ushort justPressed = (ushort)(currentButtons & ~_lastButtons);
            _lastButtons = currentButtons;

            RaiseIfPressed(justPressed, ButtonLeftShoulder, GamepadButton.LeftShoulder);
            RaiseIfPressed(justPressed, ButtonRightShoulder, GamepadButton.RightShoulder);
            RaiseIfPressed(justPressed, ButtonA, GamepadButton.A);
            RaiseIfPressed(justPressed, ButtonB, GamepadButton.B);

            bool leftTriggerJustPressed = _lastLeftTrigger < TriggerThreshold
                && state.Gamepad.LeftTrigger >= TriggerThreshold;
            bool rightTriggerJustPressed = _lastRightTrigger < TriggerThreshold
                && state.Gamepad.RightTrigger >= TriggerThreshold;

            _lastLeftTrigger = state.Gamepad.LeftTrigger;
            _lastRightTrigger = state.Gamepad.RightTrigger;

            if (leftTriggerJustPressed)
            {
                ButtonPressed?.Invoke(this, GamepadButton.LeftTrigger);
            }

            if (rightTriggerJustPressed)
            {
                ButtonPressed?.Invoke(this, GamepadButton.RightTrigger);
            }
        }

        private void RaiseIfPressed(ushort justPressed, ushort mask, GamepadButton button)
        {
            if ((justPressed & mask) != 0)
            {
                ButtonPressed?.Invoke(this, button);
            }
        }

        private void ThrowIfDisposed()
        {
            ObjectDisposedException.ThrowIf(_isDisposed, this);
        }

        public void SetVibration(ushort leftMotor, ushort rightMotor)
        {
            if (_isDisposed || !_isConnected)
            {
                return;
            }

            XInputVibration vibration = new()
            {
                LeftMotorSpeed = leftMotor,
                RightMotorSpeed = rightMotor
            };

            XInputSetState(0, ref vibration);
        }

        public async Task PulseAsync(ushort leftMotor, ushort rightMotor, int durationMs)
        {
            if (_isDisposed || !_isConnected)
            {
                return;
            }

            int safeDurationMs = Math.Max(0, durationMs);
            SetVibration(leftMotor, rightMotor);

            try
            {
                await Task.Delay(safeDurationMs);
            }
            finally
            {
                SetVibration(0, 0);
            }
        }

        public async Task DoublePulseAsync(ushort leftMotor, ushort rightMotor, int pulseDurationMs = 120, int gapMs = 120)
        {
            if (_isDisposed || !_isConnected)
            {
                return;
            }

            int safeGapMs = Math.Max(0, gapMs);
            await PulseAsync(leftMotor, rightMotor, pulseDurationMs);
            await Task.Delay(safeGapMs);
            await PulseAsync(leftMotor, rightMotor, pulseDurationMs);
        }

        [DllImport("xinput1_4.dll", EntryPoint = "XInputGetState", CallingConvention = CallingConvention.StdCall)]
        private static extern int XInputGetState(int dwUserIndex, out XInputState pState);

        [DllImport("xinput1_4.dll", EntryPoint = "XInputSetState", CallingConvention = CallingConvention.StdCall)]
        private static extern int XInputSetState(int dwUserIndex, ref XInputVibration pVibration);

        [StructLayout(LayoutKind.Sequential)]
        private struct XInputState
        {
            public int PacketNumber;
            public XInputGamepad Gamepad;
        }

        [StructLayout(LayoutKind.Sequential)]
        private struct XInputGamepad
        {
            public ushort Buttons;
            public byte LeftTrigger;
            public byte RightTrigger;
            public short ThumbLX;
            public short ThumbLY;
            public short ThumbRX;
            public short ThumbRY;
        }

        [StructLayout(LayoutKind.Sequential)]
        private struct XInputVibration
        {
            public ushort LeftMotorSpeed;
            public ushort RightMotorSpeed;
        }
    }
}
